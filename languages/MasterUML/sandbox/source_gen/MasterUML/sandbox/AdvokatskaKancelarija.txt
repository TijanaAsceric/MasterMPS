CREATE DATABASE AdvokatskaKancelarija;

CREATE TABLE Klijent (
klijentID auto,
neko VARCHAR (255),
imeKlijenta VARCHAR (255),
prezimeKlijenta VARCHAR (255),
email VARCHAR (255))

CREATE TABLE Zaposleni (
zaposleniID autoincrement,
imeZaposlenog VARCHAR (255),
prezimeZaposlenog VARCHAR (255),
email VARCHAR (255),
datumZaposlenja VARCHAR (255))

CREATE TABLE Advokat (
datumPravosudnogIspita date)

CREATE TABLE Pripravnik (
brojPreostalihIspita int)

CREATE TABLE Mesto (
mestoID autoincrement,
nazivMesta VARCHAR (255))

CREATE TABLE Predmet (
predmetID int,
nazivPredmeta VARCHAR (255),
datumPredmeta date,
pravniOsnov VARCHAR (255))

CREATE TABLE Rociste (
rocisteID int,
datumRocista date,
vremeRocista VARCHAR (255))

CREATE TABLE StavkaRocista (
rbrID int,
iznos double)

CREATE TABLE Usluga (
uslugaID int,
nazivUsluge VARCHAR (255),
opisUsluge VARCHAR (255),
cenaUsluge double)

CREATE TABLE Sud (
sudID int,
nazivSuda VARCHAR (255),
adresa VARCHAR (255))
